CREATE TABLE clients(
	id INT NOT NULL PRIMARY KEY IDENTITY,
	name VARCHAR (100) NOT NULL,
	email VARCHAR (150) NOT NULL UNIQUE,
	phone VARCHAR (20) NULL,
	created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE monetaryDonations(
	monetaryid INT NOT NULL PRIMARY KEY IDENTITY,
	date DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
	amount VARCHAR (150) NOT NULL,
	donar VARCHAR (20) NULL,
);

CREATE TABLE goodsDonations(
	goodsid INT NOT NULL PRIMARY KEY IDENTITY,
	date DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
	category VARCHAR (150) NOT NULL,
	description VARCHAR (150) NULL,
	donar VARCHAR (20) NOT NULL
);

CREATE TABLE Disaster(
	disasterid INT NOT NULL PRIMARY KEY IDENTITY,
	startdate DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
	enddate DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
	location VARCHAR (150) NOT NULL,
	description VARCHAR (150) NULL,
	aidtype VARCHAR (150) NOT NULL,
);
